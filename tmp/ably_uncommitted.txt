+// Hook for monitoring creator presence
+export function useCreatorPresence(creatorUsername: string) {
+  const { isConnected, client } = useAbly();
+  const [isLive, setIsLive] = useState(false);
+
+  useEffect(() => {
+    if (!isConnected || !client || !creatorUsername) return;
+
+    const normalizedName = `broadcast:room-${creatorUsername.toLowerCase()}`;
+    const channel = client.channels.get(normalizedName);
+    
+    console.log(`[Presence] Monitoring channel: ${normalizedName}`);
+    const checkPresence = async () => {
+      try {
+        const members = await channel.presence.get();
+        console.log(`[Presence] Members in ${channel.name}:`, members.map(m => ({ id: m.clientId, data: m.data })));
+        // Check for specific host role
+        const hostLive = members.some(m => m.data?.role === 'host');
+        console.log(`[Presence] Host role detected:`, hostLive);
+        setIsLive(hostLive);
+      } catch (e) {
+        console.error('[Presence] Failed to get presence:', e);
+      }
+    };
+
+    checkPresence();
+
+    const handlePresence = () => {
+      checkPresence();
+    };
+
+    channel.presence.subscribe(['enter', 'leave', 'present', 'update'], handlePresence);
+
+    return () => {
+      channel.presence.unsubscribe(handlePresence);
+    };
+  }, [isConnected, client, creatorUsername]);
+
+  return isLive;
+}
